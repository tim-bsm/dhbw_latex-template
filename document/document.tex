%!TEX root = ./document.tex

\input{../config/config.tex}

\begin{document}
    %%%%%%%%%% Pages before the chapters %%%%%%%%%%
    \setpagestylepre

    % Set distance between chapter title and top of page
    \RedeclareSectionCommand[beforeskip=\chapterdistance]{chapter}
    
    \ifshowtodonotes
        % Add ToDo-List
        \listoftodos
    \else
        % Disable todo notes
        \presetkeys{todonotes}{disable}{}
    \fi

    % Cover sheet
    \begin{spacing}{1}
        \input{../config/additionals/cover_sheet.tex}
    \end{spacing}

    \newpage
    \pagenumbering{Roman}

    % Blocking note
    \ifblockingnote
        \input{../config/additionals/blocking_note.tex}
        \newpage
    \fi

    % Self-Declaration
    \ifdeclaration
        \input{../config/additionals/declaration.tex}
        \newpage
    \fi

    % Abstract
    \ifabstractde
        \input{../config/additionals/abstract.tex}
        \newpage
    \else
        \ifabstracten
            \input{../config/additionals/abstract.tex}
            \newpage
        \fi
    \fi

    % Put toc and lists inside a group to deactivate the \cite command for the lists
    {
        % Deactivate \cite command
        \renewcommand{\cite}[2][]{\ignorespaces}%


        % Table of contents
        \iftableofcontents
            \input{../config/additionals/toc.tex}
            \newpage
        \fi

        % List of acronyms
        \ifacronyms
            \cleardoublepage
            \input{../config/additionals/acronyms.tex}
        \fi

        % List of images
        \ifimagelist
            \cleardoublepage
            \listoffigures
        \fi

        % List of tables
        \iftablelist
            \cleardoublepage
            \listoftables
        \fi

        % List of symbols
        \ifsymbols
            \cleardoublepage
            \input{../config/additionals/symbols.tex}
        \fi

        % Formulary
        \ifformulary
            \cleardoublepage
            \listofformels
        \fi

        % List of code
        \ifcodelist
            \cleardoublepage
            \lstlistoflistings
        \fi
    }

    \label{endOfRomanNumbering}
    
        
    %%%%%%%%%% Pages with chapters %%%%%%%%%%
    
    \cleardoublepage
    \setpagestylechapter

    % Inhalt
    \foreach \i in {01,02,03,04,05,06,07,08,09,...,99} {
        \edef\FileName{../content/chapters/chapter-\i .tex}
        \IfFileExists{\FileName}{
            \input{\FileName}
        }
    }

    \label{endOfArabicNumbering}
    \clearpage
    

    %%%%%%%%%% Pages after the chapters %%%%%%%%%%

    \input{../config/additionals/appendix.tex}
\end{document}
