\newcommand\saveequation[3]{%

    \@namedef{desc@#1}{#3}%
}

\newcommand\useequation[1]{%s
    \begin{equation}
        \formelentry{\@nameuse{desc@#1}}      % Zu Formelverzeichnis hinzuf√ºgen
        \@nameuse{equation@#1}
        \label{eq:#1} % Label setzen
    \end{equation}
    \begin{center}
        Formel \ref{eq:#1}: \@nameuse{desc@#1}
    \end{center}
}
