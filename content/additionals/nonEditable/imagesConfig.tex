    \@namedef{img@#1}{
        \begin{figure}[#2]
            \centering
            \includegraphics[#3]{#4}
            \caption{#5}
            \label{img:#1}
        \end{figure}
    }
}

% \useimage{<ID>}
\newcommand\useimage[1]{%

}
