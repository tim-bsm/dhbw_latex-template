\makeatletter

% \saveimage[OPTIONAL: <linewidth=0px, padding=0mm>]{<ID>}{<imageSetting(e.g. !ht)>}{<graphicSettings(e.g. width=0.7)>}{<imageName>}{<description>} -> OPTIONAL um grauen Rahmen um Bild zu platzieren
\newcommand\saveimage[6][]{%
    \@namedef{img@#2}{
        \begin{figure}[#3]
            \centering
            \ifthenelse{\isempty{#1}}
                {\includegraphics[#4]{#5}}%True
                {\gradientframe[#1]{\includegraphics[#4]{#5}}}%False
            \caption{#6}
            \label{img:#2}
        \end{figure}
    }
}

% \useimage{<ID>}
\newcommand\useimage[1]{%
    \@nameuse{img@#1}
}

\makeatother
